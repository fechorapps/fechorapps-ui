## Importación

```typescript
import { UiListboxComponent, type ListboxOption, type ListboxOptionGroup } from '@appcordion/ui';
```

## Uso Básico

```typescript
options: ListboxOption<string>[] = [
  { label: 'Nueva York', value: 'ny' },
  { label: 'Los Angeles', value: 'la' },
  { label: 'Chicago', value: 'chi' },
];
```

```html
<ui-listbox
  [options]="options"
  [(ngModel)]="selected"
  label="Seleccionar Ciudad"
></ui-listbox>
```

## Opciones Agrupadas

```typescript
groups: ListboxOptionGroup<string>[] = [
  {
    label: 'Norteamérica',
    options: [
      { label: 'Estados Unidos', value: 'us' },
      { label: 'Canadá', value: 'ca' },
    ],
  },
  {
    label: 'Sudamérica',
    options: [
      { label: 'Brasil', value: 'br' },
      { label: 'Argentina', value: 'ar' },
    ],
  },
];
```

```html
<ui-listbox
  [optionGroups]="groups"
></ui-listbox>
```
