## Importación

```typescript
import { UiSelectButtonComponent, type SelectButtonOption } from '@appcordion/ui';
```

## Uso Básico

```typescript
options: SelectButtonOption<string>[] = [
  { label: 'Opción 1', value: 'opt1' },
  { label: 'Opción 2', value: 'opt2' },
  { label: 'Opción 3', value: 'opt3' },
];
```

```html
<ui-select-button
  [options]="options"
  [(ngModel)]="selected"
></ui-select-button>
```
